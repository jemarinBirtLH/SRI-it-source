Comandos base recomendados

# Configuración inicial git config --global user.name "Tu Nombre" git config --global user.email "tu@email.com" # Crear repositorio y primer commit git init git add . git commit -m "Primer commit" # Conectar con GitHub y subir git remote add origin https://github.com/usuario/repositorio.git git branch -M main git push -u origin main # Flujo diario git checkout -b feature/login git add . git commit -m "feat: login base" git push -u origin feature/login
